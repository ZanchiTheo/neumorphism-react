### Displays a basic div ###

```jsx { "props": { "style": { "backgroundColor": "#929292", "textAlign": "center" } } }
import { NeuDiv } from "react-neucomponents";
  
<NeuDiv width={150} height={100} />
```

### Displays a reverted div ###

```jsx { "props": { "style": { "backgroundColor": "#929292", "textAlign": "center" } } }
import { NeuDiv } from "react-neucomponents";
  
<NeuDiv width={150} height={100} revert />
```
